104 bool near(const float lhs, const float rhs) const

115 void equals(const gu_camera_pivot lhs, const gu_camera_pivot rhs)
116 {
117 ASSERT_TRUE(near(lhs.pitch, rhs.pitch));
118 ASSERT_TRUE(near(lhs.yaw, rhs.yaw));
119 ASSERT_EQ(lhs.numCameras, rhs.numCameras);
120 for (int i = 0; i < lhs.numCameras; i++)
121 {
122 const gu_camera lcamera = lhs.cameras[i];
123 const gu_camera rcamera = rhs.cameras[i];
124 assert_near(lcamera.height, rcamera.height, 0.0001f);
125 assert_near(lcamera.height, rcamera.height, 0.0001f);
126 assert_near(lcamera.centerOffset, rcamera.centerOffset, 0.0001f);
127 assert_near(lcamera.vDirection, rcamera.vDirection, 0.0001f);
128 assert_near(lcamera.vFov, rcamera.vFov, 0.0001f);
129 assert_near(lcamera.hFov, rcamera.hFov, 0.0001f);
130 assert_near(lhs.cameraHeightOffsets[i], rhs.cameraHeightOffsets[i], 0.0001f);
131 }
132 }
