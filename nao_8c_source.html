69bool gu_nao_head_sensors_equals(const gu_nao_head_sensors lhs, const gu_nao_head_sensors rhs)
70{
71 return lhs.touchFront == rhs.touchFront
72 && lhs.touchMiddle == rhs.touchMiddle
73 && lhs.touchRear == rhs.touchRear;

76bool gu_nao_head_equals(const gu_nao_head lhs, const gu_nao_head rhs)
77{
78 return gu_pitch_yaw_joint_equals(lhs.neck, rhs.neck, 0.0001f)
79 && gu_nao_head_sensors_equals(lhs.buttons, rhs.buttons);

96{
97 return gu_pitch_roll_joint_equals(lhs.shoulder, rhs.shoulder, 0.0001f)

100 && gu_nao_hand_sensors_equals(lhs.hand, rhs.hand);

104{

119bool gu_nao_equals(const gu_nao lhs, const gu_nao rhs)
120{
121 return lhs.playerNumber == rhs.playerNumber
122 && gu_optional_field_coordinate_equals(lhs.fieldPosition, rhs.fieldPosition)
123 && gu_optional_ball_position_equals(lhs.ballPosition, rhs.ballPosition, 0.001f)
124 && gu_nao_joints_equals(lhs.joints, rhs.joints)
125 && gu_soccer_locations_equals(lhs.locations, rhs.locations);
126}

226void gu_nao_update_from_wb_types(gu_nao * nao, const gu_nao_wb_types types)
227{
228 // Player Data
229 nao->playerNumber = types.playerNumber;
230 // Head
231 nao->joints.head.neck.pitch = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.HeadPitch));
232 nao->joints.head.neck.yaw = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.HeadYaw));
233 nao->joints.head.buttons.touchFront = types.headSensors.Head_Touch_Front;
234 nao->joints.head.buttons.touchMiddle = types.headSensors.Head_Touch_Middle;
235 nao->joints.head.buttons.touchRear = types.headSensors.Head_Touch_Rear;
236 // Field Position
237 nao->fieldPosition.has_value = types.topParticles.particles[0].confidence > 0.6f;
238 nao->fieldPosition.value.position.x = cm_t_to_mm_t(i16_to_cm_t(types.topParticles.particles[0].position.x));
239 nao->fieldPosition.value.position.y = cm_t_to_mm_t(i16_to_cm_t(types.topParticles.particles[0].position.y));
240 nao->fieldPosition.value.heading = i16_to_deg_t(types.topParticles.particles[0].headingInDegrees);
241 // Ball Position
242 nao->ballPosition.has_value = wb_ball_position_confidence_percent(types.ballPosition) > 0.6;
243 nao->ballPosition.value.position.x = i16_to_cm_t(types.ballPosition.x);
244 nao->ballPosition.value.position.y = i16_to_cm_t(types.ballPosition.y);
245 nao->ballPosition.value.orientation.pitch = i16_to_deg_f(types.ballPosition.pitchInDegrees);
246 nao->ballPosition.value.orientation.yaw = i16_to_deg_f(types.ballPosition.yawInDegrees);
247 nao->ballPosition.value.orientation.roll = i16_to_deg_f(types.ballPosition.rollInDegrees);
248 // Left Arm
249 nao->joints.leftArm.shoulder.pitch = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.LShoulderPitch));
250 nao->joints.leftArm.shoulder.roll = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.LShoulderRoll));
251 nao->joints.leftArm.elbow.yaw = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.LElbowYaw));
252 nao->joints.leftArm.elbow.roll = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.LElbowRoll));
253 nao->joints.leftArm.wrist.yaw = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.LWristYaw));
254 nao->joints.leftArm.hand.touchLeft = types.handSensors.LHand_Touch_Left;
255 nao->joints.leftArm.hand.touchBack = types.handSensors.LHand_Touch_Back;
256 nao->joints.leftArm.hand.touchRight = types.handSensors.LHand_Touch_Right;
257 // Right Arm
258 nao->joints.rightArm.shoulder.pitch = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.RShoulderPitch));
259 nao->joints.rightArm.shoulder.roll = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.RShoulderRoll));
260 nao->joints.rightArm.elbow.yaw = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.RElbowYaw));
261 nao->joints.rightArm.elbow.roll = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.RElbowRoll));
262 nao->joints.rightArm.wrist.yaw = rad_f_to_deg_f(f_to_rad_f(types.torsoSensors.RWristYaw));
263 nao->joints.rightArm.hand.touchLeft = types.handSensors.RHand_Touch_Left;
264 nao->joints.rightArm.hand.touchBack = types.handSensors.RHand_Touch_Back;
265 nao->joints.rightArm.hand.touchRight = types.handSensors.RHand_Touch_Right;
266 // Left Leg
267 nao->joints.leftLeg.hip.pitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.LHipPitch));
268 nao->joints.leftLeg.hip.yawPitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.LHipYawPitch));
269 nao->joints.leftLeg.hip.roll = rad_f_to_deg_f(f_to_rad_f(types.legSensors.LHipYawPitch));
270 nao->joints.leftLeg.knee.pitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.LKneePitch));
271 nao->joints.leftLeg.ankle.pitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.LAnklePitch));
272 nao->joints.leftLeg.ankle.roll = rad_f_to_deg_f(f_to_rad_f(types.legSensors.LAnkleRoll));
273 // Right Leg
274 nao->joints.rightLeg.hip.pitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.RHipPitch));
275 nao->joints.rightLeg.hip.yawPitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.RHipYawPitch));
276 nao->joints.rightLeg.hip.roll = rad_f_to_deg_f(f_to_rad_f(types.legSensors.RHipYawPitch));
277 nao->joints.rightLeg.knee.pitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.RKneePitch));
278 nao->joints.rightLeg.ankle.pitch = rad_f_to_deg_f(f_to_rad_f(types.legSensors.RAnklePitch));
279 nao->joints.rightLeg.ankle.roll = rad_f_to_deg_f(f_to_rad_f(types.legSensors.RAnkleRoll));
280 // Locations
281 nao->locations.ball = wb_location_to_opt_rr_coord(types.ballLocation, 50);
282 nao->locations.leftGoalPost = wb_location_to_opt_rr_coord(types.leftGoalPostLocation, 50);
283 nao->locations.rightGoalPost = wb_location_to_opt_rr_coord(types.rightGoalPostLocation, 50);
284 nao->locations.goal = wb_location_to_opt_rr_coord(types.goalLocation, 50);
285 // Vision Sightings
286 nao->topCameraSightings.ball = wb_vision_detection_ball_to_opt_ellipse_sighting(types.ballSightings.balls[Top], types.ballSightings.res_width, types.ballSightings.res_height);
287 nao->bottomCameraSightings.ball = wb_vision_detection_ball_to_opt_ellipse_sighting(types.ballSightings.balls[Bottom], types.ballSightings.res_width, types.ballSightings.res_height);
288 gu_soccer_sightings_update_from_wb_vision_detection_goal(&nao->topCameraSightings, types.goalSightings.goals[Top], types.goalSightings.res_width, types.goalSightings.res_height);
289 gu_soccer_sightings_update_from_wb_vision_detection_goal(&nao->bottomCameraSightings, types.goalSightings.goals[Bottom], types.goalSightings.res_width, types.goalSightings.res_height);
290 nao->topCameraSightings.horizon = wb_vision_detection_horizon_to_opt_horizon_sighting(types.horizonSightings.horizons[Top], types.horizonSightings.res_width, types.horizonSightings.res_height);
291 nao->bottomCameraSightings.horizon = wb_vision_detection_horizon_to_opt_horizon_sighting(types.horizonSightings.horizons[Bottom], types.horizonSightings.res_width, types.horizonSightings.res_height);
292 gu_soccer_sightings_update_from_wb_vision_line_array(&nao->topCameraSightings, types.topLineSightings.lines, (int) types.topLineSightings.numLines, types.topLineSightings.res_width, types.topLineSightings.res_height);
293 gu_soccer_sightings_update_from_wb_vision_line_array(&nao->bottomCameraSightings, types.bottomLineSightings.lines, (int) types.bottomLineSightings.numLines, types.bottomLineSightings.res_width, types.bottomLineSightings.res_height);
294}
